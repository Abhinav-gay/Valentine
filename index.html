noBtn.addEventListener("mouseenter", () => { const btnWidth = noBtn.offsetWidth; const btnHeight = noBtn.offsetHeight; const padding = 20; const maxX = window.innerWidth - btnWidth - padding; // Restrict vertical range to avoid top 40% where text lives const minY = window.innerHeight * 0.45; // start below 45% height const maxY = window.innerHeight - btnHeight - padding; let newX = Math.random() * maxX + padding / 2; let newY = Math.random() * (maxY - minY) + minY; noBtn.style.left = newX + "px"; noBtn.style.top = newY + "px"; noBtn.style.transform = "translate(0, 0)"; });
